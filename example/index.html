Open the developer console and play around with the library object. Examples: 
library.location = 'Berlin, Germany';
library.books.push(Avers.mk(Book, { title: 'A Tale of Two Cities' });
library.books[0].title = 'Le Petit Prince';
Avers.attachChangeListener(library, function(path, value) { console.log(path, value); });
Avers.toJSON(library);
